logback

# 版本

logback-core-1.2.11.jar

# 问题

logback归档文件名，只能到“小时”，不支持 带"时分"格式；

举例：

xx_yyyyMMdd_HH_%i.log.gz (√ )

xx_yyyyMMdd_HH_%i.log.gz (×)

# 解决

logback支持自定义 归档策略，故基于、、SizeAndTimeBasedRollingPolicy策略重写，进行支持；

# 1.自定义归档策略 HMSRollingPolicy

package com.stock.framework.logger;

import ch.qos.logback.core.rolling.SizeAndTimeBasedFNATP;
import ch.qos.logback.core.rolling.TimeBasedRollingPolicy;
import ch.qos.logback.core.util.FileSize;
import com.stock.framework.util.UnConfusableInterface;

import java.io.File;
import java.time.Instant;
import java.time.LocalDateTime;
import java.time.LocalTime;
import java.time.ZoneId;
import java.util.regex.Pattern;


/**
 * 类介绍：满足logback.xml中 fileSizePerCategory 配置的大小后，压缩文件名中添加当前【时/分/秒】进行存档（通过fileNamePattern配置精度到为时/分/秒）
 *
 * 类使用：logback.xml配置示例如下：
 * <appender name="demo_appender" class="ch.qos.logback.core.rolling.RollingFileAppender">
 * 		<file>${baseDir}/message/mq_api_all.log</file> <!--修改点：实际日志文件名-->
 * 		<encoder>
 * 			<pattern>${globalPatternLayout}</pattern>
 * 		</encoder>
 * 		<rollingPolicy class="com.croot.HMSRollingPolicy">  <!--修改点：使用自定义滚动策略-->
 * 			<!-- 每天日志归档路径以及格式 -->
 * 			<fileNamePattern>${baseDir}/message/%d{yyyy,aux}/mq_api_all-%d{yyyyMMdd_HHmmss}-%i.log.gz</fileNamePattern> <!--修改点：如果不需要压缩，直接去掉压缩方式即可-->
 * 			<maxFileSize>${fileSizePerCategory}</maxFileSize>  <!--修改点：必填项-->
 * 			<!--日志文件保留天数-->
 * 			<maxHistory>${maxHistory}</maxHistory> <!--修改点：单位：天；如果不配置-则日志文件一直保存-->
 * 			<!--启动时清理日志文件  此项置灰清理超过保留天数的  也会清理超过总大小的-->
 * 			<cleanHistoryOnStart>false</cleanHistoryOnStart>
 * 		</rollingPolicy>
 * 	</appender>
 *
 * 类关系：
 * 源码：SizeAndTimeBasedRollingPolicy(extend TimeBasedRollingPolicy) 使用 SizeAndTimeBasedFNATP(extend TimeBasedFileNamingAndTriggeringPolicyBase)
 * 改造：HMSRollingPolicy(extend TimeBasedRollingPolicy) 使用 HMSTriggeringPolicy(extend SizeAndTimeBasedFNATP)
 *
 * @param <E>
 */
public class HMSRollingPolicy<E> extends TimeBasedRollingPolicy<E> implements UnConfusableInterface {


    FileSize maxFileSize;//logback.xml中配置的 <maxFileSize>
    private static Pattern hourFormatPattern = Pattern.compile(".*\\{.*HH.*\\}.*");
    private static Pattern minuteFormatPattern = Pattern.compile(".*\\{.*mm.*\\}.*");
    private static Pattern secondFormatPattern = Pattern.compile(".*\\{.*ss.*\\}.*");

    @Override
    public void start() {
        HMSTriggeringPolicy<E> sizeAndTimeBasedFNATP = new HMSTriggeringPolicy<E>();
        if(maxFileSize == null) {
            addError("maxFileSize property is mandatory.");
            return;
        } else {
            addInfo("Archive files will be limited to ["+maxFileSize+"] each.");
        }

        sizeAndTimeBasedFNATP.setMaxFileSize(maxFileSize);
        super.setTimeBasedFileNamingAndTriggeringPolicy(sizeAndTimeBasedFNATP);
        /*
        日志最大保存时间（logback.xml配置项maxHistory单位为：天；）
        此处需要根据logback.xml中<FileNamePattern>配置的时间单位，转为对应的时/分/秒
         */
        if (super.getMaxHistory() != 0) {
            if (secondFormatPattern.matcher(super.getFileNamePattern()).find()) {
                super.setMaxHistory(super.getMaxHistory()*24*60*60);
            } else if (minuteFormatPattern.matcher(super.getFileNamePattern()).find()) {
                super.setMaxHistory(super.getMaxHistory()*24*60);
            } else if (hourFormatPattern.matcher(super.getFileNamePattern()).find()) {
                super.setMaxHistory(super.getMaxHistory()*24);
            }
        }//super.getMaxHistory()==0时，则日志文件永久保留；

        if(!isUnboundedTotalSizeCap() && totalSizeCap.getSize() < maxFileSize.getSize()) {
            addError("totalSizeCap of ["+totalSizeCap+"] is smaller than maxFileSize ["+maxFileSize+"] which is non-sensical");
            return;
        }
        // most work is done by the parent
        super.start();
    }

    //logback启动，初始化时用到
    public void setMaxFileSize(FileSize aMaxFileSize) {
        this.maxFileSize = aMaxFileSize;
    }

    @Override
    public String toString() {
        return "c.q.l.core.rolling.HMSRollingPolicy@"+this.hashCode();
    }

}

/**
 * 类介绍：触发归档，触发策略；
 * 核心方法isTriggeringEvent：先判断大小是否满足logback.xml配置的参数 fileSizePerCategory
 * @param <E>
 */
class HMSTriggeringPolicy<E> extends SizeAndTimeBasedFNATP<E> implements UnConfusableInterface{
    FileSize maxFileSize;

    private long nextDayStartTimestamp;

    @Override
    public void start() {
        super.setMaxFileSize(maxFileSize);
        super.start();
    }

    public boolean isTriggeringEvent(File activeFile, final E event) {

        //跨天需要创建新日志
        long currentTimestamp = getCurrentTime();
        if (currentTimestamp >= nextDayStartTimestamp) {
            refreshNextDayStartTimestamp(currentTimestamp);
            return super.isTriggeringEvent(activeFile, event);
        }

        if (activeFile.length() >= maxFileSize.getSize()) {
            return super.isTriggeringEvent(activeFile, event);
        }
        return false;
    }

    public void setMaxFileSize(FileSize maxFileSize) {
        this.maxFileSize = maxFileSize;
    }

    private void refreshNextDayStartTimestamp(long currentTimestamp) {
        Instant instant = Instant.ofEpochMilli(currentTimestamp);

        LocalDateTime localDateTime = LocalDateTime.ofInstant(instant, ZoneId.systemDefault());

        LocalDateTime nextDayStart = localDateTime.plusDays(1).with(LocalTime.MIN);

        nextDayStartTimestamp =
                nextDayStart.atZone(ZoneId.systemDefault()).toInstant().toEpochMilli();
    }

}

# 2. logback.xml中使用自定义策略

	<appender name="TEST_APPENDER" class="ch.qos.logback.core.rolling.RollingFileAppender">
		<file>${baseDir}/test/test.log</file>
		<encoder>
			<pattern>[%p][%d{yyyy-MM-dd HH:mm:ss.SSS}][%t] %m%n%rEx</pattern>
		</encoder>
		<rollingPolicy class="com.stock.framework.logger.HMSRollingPolicy">
			<!-- 每天日志归档路径以及格式 -->
			<fileNamePattern>${baseDir}/test/%d{yyyy-MM-dd,aux}/test_%d{yyyyMMdd_HHmm}_%i.log.gz</fileNamePattern>
			<maxFileSize>${fileSizePerCategory}</maxFileSize>
			<!--启动时清理日志文件  此项置灰清理超过保留天数的  也会清理超过总大小的-->
			<cleanHistoryOnStart>false</cleanHistoryOnStart>
		</rollingPolicy>
		<filter class="ch.qos.logback.classic.filter.ThresholdFilter">
			<level>debug</level>
		</filter>
	</appender>
